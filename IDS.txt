			node = frontier[0]

					lista[child[2]].move = child
					if not test_goal(lista[child[2]]):
						lista[child[2]].cor = 1
						bisect.insort(frontier, lista[child[2]])
					else:
						goal_state = lista[child[2]]

			print(node.state)
			node.cor = 2
			frontier.remove(node)

		frontier = []
		resetar_lista(lista)
		deep = deep + 1
